package ru.nsu.dunaev;

// Class for Variable expression
public class Variable extends Expression {
    private final String name;

    public Variable(String name) {
        this.name = name;
    }

    @Override
    public String toString() {
        return name;
    }

    @Override
    public Expression derivative(String var) {
        return new Number(1); // Derivative of x is 1
    }

    @Override
    public int eval(String variables) {
        String[] vars = variables.split(";"); // Get the value of the variable
        for (String v : vars) {
            if (v.startsWith(name + " = ")) {
                return Integer.parseInt(v.substring(name.length() + 3)); // Return the value
            }
        }
        return 0; // If the variable is not defined, return 0
    }
}
